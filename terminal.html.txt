  <script>
    const input = document.getElementById('commandInput');
    const output = document.getElementById('output');

    input.addEventListener('keydown', function(event) {
      if (event.key === 'Enter') {
        const command = input.value.trim();
        output.innerText += `\nkenanpoculu@terminal:~$ ${command}`;

        if (command === 'open site') {
          output.innerText += `\nSite açılıyor...`;
          window.location.href = 'https://kenanpoculu.github.io/ybs241-odev/';
        } else {
          output.innerText += `\nBilinmeyen komut: ${command}`;
        }

        input.value = '';
      }
    });
  </script>
